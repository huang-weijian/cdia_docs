# 组件通用样式

{
  "style": {
    // 组件尺寸，单位 px
    "width": 0,
    "height": 0,
    // todo 组件模型
    // 组件圆角，单位 px
    "radius": 0,
    //  组件样式过渡，参考 css
    "transition": {
      // ...
    },
    // 组件视觉，参考 css
    "transform": {
      // ...
    },
    // 组件定位，absolute：相对于父组件定位，fixed：相对于表盘定位
    "position": "absolute",
    // 组件定位（相对于父组件或表盘），单位 px
    "top": 0,
    "right": 0,
    "bottom": 0,
    "left": 0,
    // 组件定位层级，不重复正整数
    "z": 1,
    // 组件动画
    "animation": {
      // 动画持续时间，单位 ms
      "duration": 0,
      "frames": {
        "0%": {
          // ...通用样式与组件特有样式
        },
        "100%": {}
      }
    }
  }
}
