# 文本组件

{
  "type": "label"
}

特有属性

{
  "attrs": {
    // 展示文本，可为 string、number
    "text": "xxx"
  }
}

特有样式

{
  "style": {
    // 十六进制或 rgb 格式颜色
    "color": "xxx",
    // 字体尺寸，单位为 px
    "fontSize": 0,
    // 字重
    "fontWeight": 600
  }
}
