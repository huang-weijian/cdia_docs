# 起步示例

下方是一个简易的表盘配置示例

{
   // 表盘系列名称
   "name":"demo_cdia",
   // 表盘形状（目标平台表盘屏幕类型）
   "shape":"rect",
   // 表盘尺寸，单位为 px
   "size":{
      "width":240,
      "height":240
   },
   // 快应用表盘配置版本
   "quickVersion":1,
   // 数据项数组（由平台提供），可在 themes 的组件内使用
   "data":["step","power","datetime","timestamp"],
   // 格式化数据项（互相之间不可引用），可在 themes 的组件内使用
   "formattedData":{
      "cnDate":{
         // 数据来源，从数据项 data 字段的数组中取，每个数据项有各自的模式匹配参数
         "origin":"datetime",
         // datetime 会将 yyyy、mm、dd进行模式匹配替换，{{}}中支持表达式运算
         "val":"{{yyyy}}年{{mm}}月{{dd}}日"
      },
   },
   // 表盘主题（实际展示），在手表不同模式下应用（常态模式、白天模式、黑夜模式、安静模式等）
   "themes":[
      {
         // 主题名称
         "name":"normal_theme",
         // 主题匹配模式
         "mode":"normal",
         // 亮屏与息屏
         "status":"",
         // 组件层级默认按组件顺序，类似 css 的 z-index
         "components":[
            {
               // 组件类型
               "type":"image",
               // 组件属性
               "attrs":{
                  // 图片绝对路径
                  "src":"/img/watch_bg.png"
               },
               // 组件样式
               "style":{
                  "width":240,
                  "height":240,
                  // 类似 css 的 top、right、bottom、left，默认值为 left:0,top:0
                  "left":0,
                  "top":0,
                  // 组件在 z 轴的层级
                  "z":1,
               }
            },
            // 日期文本
            {
               "type":"label",
               "attrs":{
                  "fontSize":16,
                  // 在数据项或格式化数据项中搜索
                  "text":"{{cnDate}}"
               },
               "style":{
                  "left":10,
                  "top":10,
                  "z":2,
               }
            },
            // 电量标识
            {
               "type":"label",
               "attrs":{
                  "fontSize":16,
                  // 在数据项或格式化数据项中搜索
                  "text":"剩余{{power}}电量{{power>50?"":"!!!"}}"
               },
               "style":{
                  "right":10,
                  "top":10,
                  "z":2,
               }
            }
         ]
      }
   ]
}
